⚠️  المشكلة:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
أنت تختار "Autoscale" بدلاً من "Reserved VM"

✅  الحل:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الطريقة 1 - إذا نشرت deployment سابق:
───────────────────────────────────────────────────────────────
1. افتح صفحة "Deployments" (أيقونة الصاروخ 🚀)
2. اضغط تبويب "Manage"
3. ابحث عن "Change deployment type"
4. اختر "Reserved VM - Web Server"
5. احفظ وأعد النشر


الطريقة 2 - إذا هذا أول deployment:
───────────────────────────────────────────────────────────────
1. اضغط زر "Deploy"
2. عندما يسألك "What type?"
3. اختر: "Reserved VM - Web Server" ✓
4. لا تختر: "Autoscale" ❌


الطريقة 3 - احذف Autoscale واعمل جديد:
───────────────────────────────────────────────────────────────
1. افتح "Deployments"
2. احذف deployment الموجود (Autoscale)
3. اضغط "New deployment"
4. اختر "Reserved VM - Web Server"
5. اضغط Deploy
